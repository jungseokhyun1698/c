#include <stdio.h>
#include <string.h>
#define TRUE 1
#define FALSE 0

struct groups {
    char *name;
    int age;
    char *gender;
    char *hobby;
};

struct groups g1[1000] = {
    {"hong", 30, "M", "soccer"},
    {"kim", 25, "M", "music"},
    {"park", 20, "F", "movie"}
};

void func1(int n)
{
    printf("%d. %s/%d/%s/%s\n", n + 1, g1[n].name, g1[n].age, g1[n].gender, g1[n].hobby);
}

void func2()
{
    int i = 0;
    while(TRUE) {
        func1(i);
        i++;
        if (i == 3) {
            break;
        }
    }
}

void func3(int n)
{
    printf("이름:");
    scanf("%ms", &g1[n].name);
    while(TRUE) {
        printf("나이:");
        scanf("%d", &g1[n].age);
        if (g1[n].age >= 0 && g1[n].age <= 1000) {
            break;
        } else {
            continue;
        }
    }
    while(TRUE) {
        printf("성별:");
        scanf("%ms", &g1[n].gender);
        if (strcmp(g1[n].gender, "F") != 0 && strcmp(g1[n].gender, "M") != 0) {
            continue;
        } else {
            break;
        }
    }
    printf("취미:");
    scanf("%ms", &g1[n].hobby);
}

int func4(int n)
{
    int z = 0;
    printf("%d명 등록. 0(continue) / 1(break) ", n + 1);
    scanf("%d", &z);
    return z != 0;
}

void main()
{
    func2();
    for (int i = 3; i < 1000; i++) {
        func3(i);
        for (int j = 0; j <= i; j++) {
            func1(j);
        }
        if (func4(i)) {
            break;
        }
    }
}
