#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void quick_sort(int num[], int start, int end) {
    if (start > end) {
        return;
    }
    int num1 = 0, num2 = 0;
    int pivot = num[start];
    int low = start + 1;
    int high = end;
    while (low <= high) {
        if (num[low] < pivot) {
            if (num[high] > pivot) {
                low++;
                high--;
            } else {
                low++;
            }
        } else if (num[high] > pivot) {
            high--;
        } else {
            num1 = num[low];
            num[low] = num[high];
            num[high] = num1;
        }
    }
    if (num[high] < num[start]) {
        num2 = num[start];
        num[start] = num[high];
        num[high] = num2;
    }
    quick_sort(num, start, high - 1);
    quick_sort(num, high + 1, end);
}

int main()
{
    int num[6];
    srand(time(NULL));
    for (int i = 0; i < 6; i++) {
        num[i] = (int)(rand() % 45) + 1;
        for (int j = 0; j < i; j++) {
            if (num[j] == num[i]) {
                i--;
            }
        }
    }
    for (int i = 0; i < 6; i++) {
        printf("정렬 전: %d ", num[i]);
    }
    printf("\n");
    quick_sort(num, 0, 5);
    for (int i = 0; i < 6; i++) {
        printf("정렬 후: %d ", num[i]);
    }
}
