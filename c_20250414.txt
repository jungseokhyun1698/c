        printf("성별:");

            continue;

            break;
        }
    }
    while(TRUE) {
        printf("취미(없으면 0):");

            while(TRUE) {
                printf("etc(없으면 0):");
                scanf("%ms", &u[n].etc);
                if (strcmp(u[n].etc, "0") == 0) {
                    continue;
                }
                break;
            }
        }
        break;
    }
}
